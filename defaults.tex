\usepackage{pgfplots}

% Create command to specify size of a plot created using
% my save_tex() command. \setfig{} sets the size of the
% plot in the macro \figurewidth. \resetfig is set to be 
% automatically used when starting a figure environment,
% setting \figurewidth to a default value of \texwidth.
\newcommand{\resetfig}{
\ifcsname figurewidth\endcsname\else
  \newlength\figurewidth
\fi
\setlength\figurewidth{\textwidth}
}
\newcommand{\setfig}[1]{
\ifcsname figurewidth\endcsname\else
  \setlength\figurewidth{#1}
\fi
\setlength\figurewidth{#1}
}

% Set figure environments to always center the image and to
% have \figurewidth set to \texwidth, in case a pgfplot
% generated by my save_tex() command is used.
\makeatletter
\g@addto@macro\@floatboxreset\centering
\g@addto@macro\@floatboxreset\resetfig
\makeatother
